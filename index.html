export class Email {
  constructor(private readonly value: string) {
    if (!this.isValid(value)) {
      throw new Error('Invalid email');
    }
  }
  
  private isValid(email: string): boolean {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
  }
}

@Injectable()
export class CreateUserUseCase {
  constructor(
    @Inject('UserRepository')
    private userRepo: UserRepository
  ) {}
  
  async execute(dto: CreateUserDto) {
    const user = new User(
      uuidv4(),
      dto.email,
      dto.name
    );
    return await this.userRepo.save(user);
  }
}

      name: user.name,
      createdAt: user.createdAt

}

Component Location Description & Code Pattern
ORM Entity src/infrastructure/persistence/{entity}/ Database schema mapping 
@Entity('users')
export class UserSchema {
  @PrimaryGeneratedColumn('uuid')
  id: string;
  
  @Column()
  email: string;
  
  @Column()
  name: string;
  
  @CreateDateColumn()
  createdAt: Date;
}
Repository Implementation src/infrastructure/persistence/{entity}/ Concrete implementation of domain repository 
@Injectable()
export class TypeOrmUserRepository 
  implements UserRepository {
  
  constructor(
    @InjectRepository(UserSchema)
    private repo: Repository<UserSchema>
  ) {}
  
  async findById(id: string): Promise<User> {
    const schema = await this.repo.findOne({
      where: { id }
    });
    return this.toDomain(schema);
  }
  
  private toDomain(schema: UserSchema): User {
    return new User(
      schema.id,
      schema.email,
      schema.name
    );
  }
}
Database Module src/infrastructure/persistence/ Database configuration 
@Module({
  imports: [
    TypeOrmModule.forFeature([UserSchema])
  ],
  providers: [
    {
      provide: 'UserRepository',
      useClass: TypeOrmUserRepository
    }
  ],
  exports: ['UserRepository']
})
export class UserPersistenceModule {}

Component Location Description & Code Pattern
Controller src/presentation/{entity}/ HTTP endpoint handlers 
@Controller('users')
export class UserController {
  constructor(
    private createUser: CreateUserUseCase,
    private getUser: GetUserUseCase
  ) {}
  
  @Post()
  async create(@Body() dto: CreateUserRequestDto) {
    const result = await this.createUser.execute(dto);
    return UserMapper.toDto(result);
  }
  
  @Get(':id')
  async getById(@Param('id') id: string) {
    const user = await this.getUser.execute(id);
    return UserMapper.toDto(user);
  }
}
Request DTO src/presentation/{entity}/dto/ HTTP request validation 
export class CreateUserRequestDto {
  @IsEmail()
  email: string;
  
  @IsString()
  @MinLength(3)
  name: string;
  
  @IsString()
  @MinLength(8)
  password: string;
}
Response DTO src/presentation/{entity}/dto/ HTTP response structure (can reuse Application DTOs) 
export class UserResponseDto {

ğŸ”„ Complete Development Flow
Step Action What to Create Dependencies
1 Define Domain Create Entity, Repository Interface, Value Objects, Exceptions None - Start here!
2 Create Use Cases Use Case classes, Input/Output DTOs, Mappers Depends on: Domain Entity, Repository Interface
3 Implement Infrastructure ORM Entity, Repository Implementation, Database Module Depends on: Repository Interface from Domain
4 Create API Controller, Request/Response DTOs, Module Depends on: Use Cases, Mappers
5 Wire Dependencies Register providers in modules, inject dependencies All layers ready
6 Test Unit tests for each layer, Integration tests for API Everything implemented
ğŸ’¡ Pro Tip: Always work from inner layers (Domain) to outer layers (Presentation). This ensures your business logic is independent of technical details.

ğŸ”„ Request Flow Example

POST /users with body: { "email": "john@example.com", "name": "John" }

  1. Controller receives request â†’ validates CreateUserRequestDto
  2. Controller calls CreateUserUseCase.execute(dto)
  3. Use Case creates domain User entity
  4. Use Case calls UserRepository.save(user)
  5. Repository converts to UserSchema and saves to database
  6. Repository returns domain User
  7. Controller uses UserMapper.toDto(user)
  8. Controller returns UserResponseDto to client
